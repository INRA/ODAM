    How to get google search results¶

    Find all the links that resulted from a keyword search on google

    • https://stackoverflow.com/questions/32889136/how-to-get-google-search-results
    library(XML)
library(RCurl)

# Build the Google Search URL
getGoogleURL <- function(search.term, domain = '.com', quotes=TRUE) 
{
    search.term <- gsub(' ', '%20', search.term)
    if(quotes) search.term <- paste('%22', search.term, '%22', sep='') 
        getGoogleURL <- paste('http://www.google', domain, '/search?q=',
        search.term, sep='')
}

# Do the search 
getGoogleLinks <- function(search.term, domain = '.com', quotes=TRUE) 
{
   google.url <- getGoogleURL(search.term=search.term, domain = domain, quotes=quotes)
   doc <- getURL(google.url, httpheader = c("User-Agent" = "R (2.10.0)"))
   html <- htmlTreeParse(doc, useInternalNodes = TRUE, error=function (...){})
   nodes <- getNodeSet(html, "//div[@class='kCrYT']//a")
   links <- sapply(nodes, function(x) x <- xmlAttrs(x)[["href"]])
   links <- links[grep(sprintf("google%s/search",domain), links, invert = TRUE)]
   gsub('/url.q=', '', gsub('&sa.+$', '', links))
}

# Search terms
search.term <- "frim1 dataset"

# Get all the links
links <- getGoogleLinks(search.term, domain = '.fr', quotes="FALSE")
links

# Select only those with a DOI
links[ nchar(gsub('^http.+$', '', gsub('^.+doi:', '', links)))>0 ]
